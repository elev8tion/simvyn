  <action>
Add a `cli(program)` method to the logViewerModule object in manifest.ts. Follow the exact pattern from app-management and location manifests.

**Command structure:**
```
simvyn logs <device> [options]
```

Options:
- `--level <level>` — Minimum log level to display (choices: verbose, debug, info, warning, error, fatal). Default: "info"
- `--filter <pattern>` — Filter log entries by regex pattern (matches against message and processName)
- `--json` — Output raw JSON instead of formatted text

**Implementation:**

```typescript
cli(program) {
  program
    .command("logs <device>")
    .description("Stream device logs in real-time")
    .option("--level <level>", "Minimum log level", "info")
    .option("--filter <pattern>", "Filter by text/regex pattern")
    .option("--json", "Output as JSON lines")
    .action(async (deviceId: string, opts: { level: string; filter?: string; json?: boolean }) => {
      // 1. Import core (dynamic, same pattern as app-management/location)
      const { createAvailableAdapters, createDeviceManager, createProcessManager } = await import("@simvyn/core");

      // 2. Create adapters and find device (with prefix matching)
      const adapters = await createAvailableAdapters();
      const dm = createDeviceManager(adapters);
      const pm = createProcessManager();

      try {
        const devices = await dm.refresh();
        const target = devices.find(d => d.id === deviceId || d.id.startsWith(deviceId));
        if (!target) {
          console.error(`Device not found: ${deviceId}`);
          process.exit(1);
        }
        if (target.state !== "booted") {
          console.error("Device must be booted");
          process.exit(1);
        }

        // 3. Import parse helpers and readline from log-streamer
        // Use the exported parse functions directly instead of full LogStreamer
        // (CLI doesn't need batching — it outputs each line immediately)
        const { createInterface } = await import("node:readline");

        // 4. Validate level option
        const levels = ["verbose", "debug", "info", "warning", "error", "fatal"];
        const minLevelIndex = levels.indexOf(opts.level);
        if (minLevelIndex === -1) {
          console.error(`Invalid level: ${opts.level}. Choose from: ${levels.join(", ")}`);
          process.exit(1);
        }

        // 5. Compile filter regex
        let filterRe: RegExp | null = null;
        if (opts.filter) {
          try {
            filterRe = new RegExp(opts.filter);
          } catch {
            console.error(`Invalid regex pattern: ${opts.filter}`);
            process.exit(1);
          }
        }

        // 6. Spawn platform-appropriate log command
        const child = target.platform === "ios"
          ? pm.spawn("xcrun", ["simctl", "spawn", target.id, "log", "stream", "--style", "ndjson", "--level", "debug"])
          : pm.spawn("adb", ["-s", target.id, "logcat", "-v", "json"]);

        const rl = createInterface({ input: child.stdout! });

        // 7. Import parse functions from log-streamer
        const { parseIosLogLine, parseAndroidLogLine } = await import("./log-streamer.js");

        // 8. Define ANSI color map for log levels
        const levelColors: Record<string, string> = {
          verbose: "\x1b[90m",   // gray
          debug:   "\x1b[36m",   // cyan
          info:    "\x1b[37m",   // white
          warning: "\x1b[33m",   // yellow
          error:   "\x1b[31m",   // red
          fatal:   "\x1b[35m",   // magenta
        };
        const reset = "\x1b[0m";

        // 9. Stream and filter
        console.error(`Streaming ${target.platform} logs from ${target.name} (level >= ${opts.level})...`);

        rl.on("line", (line) => {
          const entry = target.platform === "ios"
            ? parseIosLogLine(line)
            : parseAndroidLogLine(line);
          if (!entry) return;

          // Level filter
          if (levels.indexOf(entry.level) < minLevelIndex) return;

          // Regex filter on message and processName
          if (filterRe && !filterRe.test(entry.message) && !filterRe.test(entry.processName)) return;

          if (opts.json) {
            console.log(JSON.stringify(entry));
          } else {
            const color = levelColors[entry.level] ?? "";
            const lvl = entry.level.toUpperCase().padEnd(7);
            console.log(`${color}${entry.timestamp} [${lvl}] ${entry.processName}: ${entry.message}${reset}`);
          }
        });

        // 10. Handle SIGINT for clean exit
        process.on("SIGINT", () => {
          child.kill("SIGTERM");
          dm.stop();
          pm.cleanup();
          process.exit(0);
        });

        // Wait for stream to end (e.g., device disconnected)
        await new Promise<void>((resolve) => {
          rl.on("close", resolve);
        });
      } finally {
        dm.stop();
        pm.cleanup();
      }
    });
}
```

The parse functions (`parseIosLogLine`, `parseAndroidLogLine`) must be exported from `log-streamer.ts` (they should be from Plan 01). If they're not standalone exports, extract them. The CLI does NOT use LogStreamer's batching — it processes each line immediately for lowest latency.

Key details:
- Use `console.error` for the "Streaming..." status message so it doesn't pollute stdout (allows piping `simvyn logs <device> > file.log`)
- ANSI colors use raw escape codes (not chalk) to avoid an import for 6 constants. Respects `NO_COLOR` env: if `process.env.NO_COLOR` is set, skip colors.
- Level filter is ordinal: verbose < debug < info < warning < error < fatal
