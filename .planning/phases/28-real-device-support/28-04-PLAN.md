**Part A: DeviceSelector grouping** (`packages/dashboard/src/components/DeviceSelector.tsx`):

1. Replace the existing `groupByPlatform()` function with a new `groupDevices()` function:
   ```typescript
   function groupDevices(devices: Device[]) {
     const groups: Record<string, Device[]> = {};
     for (const d of devices) {
       let section: string;
       if (d.deviceType === "Physical") {
         section = "Physical Devices";
       } else if (d.platform === "android") {
         section = "Emulators";
       } else {
         section = "Simulators";
       }
       (groups[section] ??= []).push(d);
     }
     // Return in fixed order: Physical first, then Simulators, then Emulators
     const ordered: Record<string, Device[]> = {};
     for (const key of ["Physical Devices", "Simulators", "Emulators"]) {
       if (groups[key]?.length) ordered[key] = groups[key];
     }
     return ordered;
   }
   ```

2. Update the component to use `groupDevices(devices)` instead of `groupByPlatform(devices)`.
