  Inside the existing `wsBroker.registerChannel("logs", ...)` callback, add two new message type handlers BEFORE the closing of the callback (after the existing `stop-stream` handler):

  **get-history handler:**
  ```typescript
  if (type === "get-history") {
    const { deviceId, before, limit } = data as { deviceId: string; before?: number; limit?: number };
    if (!deviceId) { sendError("Missing deviceId"); return; }

    const history = getLogHistory(deviceId); // chronological order (oldest first)
    const pageSize = Math.min(limit ?? 500, 500);
    const endIdx = before != null ? Math.min(before, history.length) : history.length;
    const startIdx = Math.max(0, endIdx - pageSize);
    const page = history.slice(startIdx, endIdx);

    wsBroker.send(socket, "logs", "history-page", {
      deviceId,
      entries: [...page].reverse(), // newest-first for client
      cursor: startIdx,
      hasMore: startIdx > 0,
    }, requestId);
    return;
  }
  ```

  The `before` parameter is a cursor (index into the server's chronological buffer). First request omits `before` (gets newest page). Subsequent requests pass `cursor` from previous response as `before`.

  **clear-device-logs handler:**
  ```typescript
  if (type === "clear-device-logs") {
    const { deviceId } = data as { deviceId: string };
    if (!deviceId) { sendError("Missing deviceId"); return; }

    const device = deviceManager.devices.find((d: Device) => d.id === deviceId);
    if (!device) { sendError(`Device not found: ${deviceId}`); return; }

    // Clear device-side logs
    if (device.platform === "android") {
      // adb logcat -c clears the Android log buffer
      processManager.spawn("adb", ["-s", deviceId, "logcat", "-c"]);
    } else if (device.platform === "ios") {
      // iOS has no equivalent â€” restart the log stream for a fresh start
      const ref = activeStreams.get(deviceId);
      if (ref && ref.streamer.isRunning) {
        ref.streamer.stop();
        ref.streamer.start();
      }
    }

    // Clear server-side history buffer
    const ref = activeStreams.get(deviceId);
    if (ref) {
      ref.streamer.clearHistory();
    }

    wsBroker.send(socket, "logs", "device-cleared", { deviceId }, requestId);
    return;
  }
  ```

  Important: `activeStreams` is already a module-level `Map<string, StreamRef>` accessible within the handler closure. The `processManager` is already destructured from `fastify` at the top of `registerLogWsHandler`. The `Device` type is already imported.
